// Force Layout Demo

function Miserables()
{    
    this.width    = 800;
    this.height   = 500;    
    this.radius   = 8;

    this.init = function(d)
    {
        this.force = d3.layout.force()
            .nodes(d.nodes)
            .links(d.links)
            .charge(-100)
            .size([this.width,this.height])
            .start()
        ;

        this.color = d3.scale.category20();
    }

    /*********************************************************************************
    <div>
        <svg width="{width}" height="{height}">
            <include template="{MiserableLinks}" data="{d.links}"/>
            <include template="{MiserableNodes}" data="{d.nodes}"/>
        </svg>
    </div>
    *********************************************************************************/
}

function MiserableNodes()
{
    this.configureDrag = function()
    {
        var force = this.parent.force;
        d3.select(redot.element).datum(this.d).call(force.drag);
    };
    
    /**********************************************
    <circle
        onenter="{configureDrag()}"
        class="node" 
        r="{parent.radius}" 
        cx="{d.x}" 
        cy="{d.y}" 
        fill="{parent.color(d.group)}" 
    />
    **********************************************/
}

function MiserableLinks()
{
    /**********************************************
    <line
        class="link"
        x1="{d.source.x}" 
        y1="{d.source.y}" 
        x2="{d.target.x}" 
        y2="{d.target.y}"
        style="stroke-width: {Math.sqrt(d.value)}"
    />
    **********************************************/
}
